		\draw[]
			(\x,2pt)--(\x,-2pt)node[below]{$\x$}
			(-\x,2pt)--(-\x,-2pt)node[below]{$-\x$}
			(2pt,\x)--(-2pt,\x)node[left]{$\x$}
		;
	\draw[,thick,blue,latex-latex]
		plot[domain=-2.236:2.236]
		(\x,{pow(\x,2)})
	;
\end{tikzpicture}
